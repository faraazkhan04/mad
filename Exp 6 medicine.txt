Mainactivity.java

package com.example.medicineremainder;

import android.database.Cursor;
import android.os.Bundle;
import android.view.View;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText medname, meddate;
    Button insert, fetch;
    Spinner day;
    Switch toggle;
    TextView medtxt;
    DataBaseConn db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        medname = findViewById(R.id.medicinename);
        meddate = findViewById(R.id.date);
        insert = findViewById(R.id.insert);
        fetch = findViewById(R.id.fetch);
        day = findViewById(R.id.spinner);
        toggle = findViewById(R.id.switch1);
        medtxt = findViewById(R.id.medtext);
        db = new DataBaseConn(this);

        fetch.setVisibility(View.INVISIBLE);

        toggle.setOnCheckedChangeListener((b, checked) -> {
            insert.setVisibility(checked ? View.INVISIBLE : View.VISIBLE);
            fetch.setVisibility(checked ? View.VISIBLE : View.INVISIBLE);
            medname.setVisibility(checked ? View.INVISIBLE : View.VISIBLE);
            medtxt.setVisibility(checked ? View.INVISIBLE : View.VISIBLE);
        });

        insert.setOnClickListener(v -> {
            if (db.insertValues(
                    medname.getText().toString().trim(),
                    meddate.getText().toString().trim(),
                    day.getSelectedItem().toString().trim()
            )) {
                Toast.makeText(this, "Data Inserted", Toast.LENGTH_LONG).show();
                medname.setText(""); meddate.setText("");
            } else Toast.makeText(this, "Data Not Inserted", Toast.LENGTH_LONG).show();
        });

        fetch.setOnClickListener(v -> {
            try (Cursor c = db.fetchData(
                    meddate.getText().toString().trim(),
                    day.getSelectedItem().toString().trim()
            )) {
                if (c != null && c.moveToFirst()) {
                    StringBuilder meds = new StringBuilder();
                    do meds.append(c.getString(c.getColumnIndexOrThrow("MedicineName"))).append('\n');
                    while (c.moveToNext());
                    Toast.makeText(this, meds.toString(), Toast.LENGTH_LONG).show();
                } else Toast.makeText(this, "No Entries in Database", Toast.LENGTH_SHORT).show();
            }
        });
    }
}



DatabaseConnection.java


package com.example.medicineremainder;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;

public class DataBaseConn extends SQLiteOpenHelper {

    public DataBaseConn(Context context) {
        super(context, "Medicinedb", null, 2);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        db.execSQL("CREATE TABLE MDTable(MedicineName TEXT, med_date TEXT, med_time TEXT)");
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        db.execSQL("DROP TABLE IF EXISTS MDTable");
        onCreate(db);
    }

    public boolean insertValues(String name, String date, String time) {
        ContentValues cv = new ContentValues();
        cv.put("MedicineName", name);
        cv.put("med_date", date);
        cv.put("med_time", time);
        return getWritableDatabase().insert("MDTable", null, cv) != -1;
    }

    public Cursor fetchData(String date, String time) {
        return getReadableDatabase().query(
                "MDTable",
                new String[]{"MedicineName"},
                "med_date=? AND med_time=?",
                new String[]{date.trim(), time.trim()},
                null, null, null
        );
    }
}
