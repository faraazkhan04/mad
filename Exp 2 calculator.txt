package com.example.calc;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText n1, n2, result;
    EditText activeField;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        n1 = findViewById(R.id.editTextNumber);
        n2 = findViewById(R.id.editTextNumber2);
        result = findViewById(R.id.editTextNumber3);

        activeField = n1;

        n1.setOnClickListener(v -> activeField = n1);
        n2.setOnClickListener(v -> activeField = n2);

        int[] numberIds = {
                R.id.btn0, R.id.btn1, R.id.btn2, R.id.btn3, R.id.btn4,
                R.id.btn5, R.id.btn6, R.id.btn7, R.id.btn8, R.id.btn9, R.id.btnDot
        };

        View.OnClickListener numberListener = v -> {
            Button b = (Button) v;
            activeField.append(b.getText().toString());
        };

        for (int id : numberIds) {
            findViewById(id).setOnClickListener(numberListener);
        }

        Button add = findViewById(R.id.button);
        Button sub = findViewById(R.id.button2);
        Button mul = findViewById(R.id.button3);
        Button div = findViewById(R.id.button4);
        Button clear = findViewById(R.id.button5);

        add.setOnClickListener(v -> calculate('+'));
        sub.setOnClickListener(v -> calculate('-'));
        mul.setOnClickListener(v -> calculate('*'));
        div.setOnClickListener(v -> calculate('/'));

        clear.setOnClickListener(v -> {
            n1.setText("");
            n2.setText("");
            result.setText("");
        });
    }

    void calculate(char op) {
        try {
            float v1 = Float.parseFloat(n1.getText().toString());
            float v2 = Float.parseFloat(n2.getText().toString());
            float res = 0;

            switch (op) {
                case '+':
                    res = v1 + v2;
                    break;
                case '-':
                    res = v1 - v2;
                    break;
                case '*':
                    res = v1 * v2;
                    break;
                case '/':
                    if (v2 == 0) {
                        result.setText("Error");
                        return;
                    }
                    res = v1 / v2;
                    break;
            }

            result.setText(String.valueOf(res));

        } catch (Exception e) {
            result.setText("Invalid");
        }
    }
}
