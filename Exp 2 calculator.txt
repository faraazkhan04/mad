package com.example.calc;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText n1, n2, result;
    Button addBtn, subBtn, mulBtn, divBtn, clearBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Linking XML IDs
        n1 = findViewById(R.id.editTextNumber);
        n2 = findViewById(R.id.editTextNumber2);
        result = findViewById(R.id.editTextNumber3);

        addBtn = findViewById(R.id.button);      // +
        subBtn = findViewById(R.id.button2);     // -
        mulBtn = findViewById(R.id.button3);     // *
        divBtn = findViewById(R.id.button4);     // /
        clearBtn = findViewById(R.id.button5);   // Clear

        // Addition
        addBtn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                float v1 = Float.parseFloat(n1.getText().toString());
                float v2 = Float.parseFloat(n2.getText().toString());
                result.setText(String.valueOf(v1 + v2));
            }
        });

        // Subtraction
        subBtn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                float v1 = Float.parseFloat(n1.getText().toString());
                float v2 = Float.parseFloat(n2.getText().toString());
                result.setText(String.valueOf(v1 - v2));
            }
        });

        // Multiplication
        mulBtn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                float v1 = Float.parseFloat(n1.getText().toString());
                float v2 = Float.parseFloat(n2.getText().toString());
                result.setText(String.valueOf(v1 * v2));
            }
        });

        // Division
        divBtn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                float v1 = Float.parseFloat(n1.getText().toString());
                float v2 = Float.parseFloat(n2.getText().toString());
                if (v2 != 0) {
                    result.setText(String.valueOf(v1 / v2));
                } else {
                    result.setText("Error");
                }
            }
        });

        // Clear
        clearBtn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                n1.setText("");
                n2.setText("");
                result.setText("");
            }
        });
    }
}
